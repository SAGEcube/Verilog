`timescale 1ns / 1ps

module pc32 (
    input clk,              // clock
    input rst,              // reset
    input [31:0] pc_next,   // next pc value from datapath/control 
    output reg [31:0] pc_q  // current pc value (output)
);

    always @(posedge clk or posedge rst) begin
        if (rst) begin 
            pc_q <= 32'b0;     // reset PC
        end else begin 
            pc_q <= pc_next;   // update PC
        end 
    end 

endmodule
